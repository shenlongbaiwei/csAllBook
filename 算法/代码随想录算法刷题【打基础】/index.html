    满二叉树

    深度为k，总结点数为2^k-1个

    完全二叉树

    除了最底层结点未填满之外，其他的都填满了，并且最底层未填满的结点都尽可能靠左。

    二叉树的存储方式¶

    二叉树可以链式存储，也可以顺序存储。

    链式存储不必多说，直接就是一个结点左右两个指针。

    顺序存储是一个数组：如果父节点的数组下标是i，那么左右子节点的下标为2*i+1,2*i+2.

    一般链式二叉树更容易理解，所以一般都用链式存储法

    二叉树的遍历¶

    我的理解就是，前序，中序，后序，层序。

    实际是主要使用了图论中的DFS和BFS，

    DFS产生了前序，中序，后序。

    BFS产生了层序。

    前序，中序，后序都有递归和迭代。

    层序只有迭代。

    这些遍历都是需要栈和队列结合的。

    ### 二叉树的迭代遍历

    用迭代法(非递归的方式)来实现二叉树的前后中序遍历

    #### 前序遍历(迭代法)

    C++
    class solution{
 public:
     vector<int> preorderTraversal(TreeNode*root){
         stack<TreeNode*> st;
         vector<int> result;
         if(root==NULL) return result;
         st.push(root);
         while(!st.empty()){
             TreeNode*node=st.top();
             result.push_back(node->val);
             if(node->right) st.push(node->right);
             if(node->left) st.push(node->left);
         }
         return result;
     }
};

    此时如果想把前序遍历代码改下顺序就得到中序遍历还是很难搞的。

    再用迭代法写中序遍历的时候，会发现套路又不一样了，目前的前序遍历的逻辑无法直接应用到中序遍历上。

    中序遍历（迭代法）¶

    1. 处理：将元素放进result数组中
    2. 访问：遍历结点
    C++
    calss solution
{
    public:
    vector<int> inorderTraversal(TreeNode*root){
        stack<TreeNode*> st;
        vector<int> result;
        TreeNode*cur=root;//定义一个指针来辅助进行栈的操作，并且初始值为根结点。
        if(root==nullptr) return result;
        //st.push(cur);这个地方不用提前入栈，因为，这个指针会在循环第一次就放入栈底，因此不用提前放，如果提前放了，那么后面循环中就会重复将根节点进栈两次，导致最后访问的时候出错。
        while(cur!=null||!st.empty()){//当当前指针和栈全为空，循环停止。
            if(cur!=nullptr){
                st.push(cur);//将当前结点进栈
                cur=cur->left;
                }
            else{
                cur=st.top();
                st.pop();
                result.push_back(cur->val);//出栈，访问将栈顶元素放进去，这个地方其实包含了先访问左节点，再访问右结点的操作，因为左节点会后进栈，先出栈，所以访问的时候永远是左节点先访问，中结点后访问，所以有个隐藏顺序，然后在访问完之后会进行相关的左根右的顺序，所以在这一步进行cur->cur->right;
                cur=cur->right;
            }
        }
        return result;
    }
};

    后序遍历（迭代法）¶

    后序遍历：左右根，这个操作跟前序相反，因此这个地方只需要在前序遍历的基础上进行修改就行。‘

    C++
    class solution{
    public:
    vector<int> postorderTraverseal(TreeNode*root){
        stack<TreeNode*> st;
        vector<int> result;
        if(root==null) return result;
        st.push(root);
        while(!st.empty()){
            TreeNode*cur=st.top();
            st.pop();
            result.push_back(cur->val);
            if(cur->right) st.push(cur->left);
            if(cur->left) st.push(cur->right);
        }
        reverse(result.begin(),result.end());
        return result;
    }
};

    思路就是前序遍历的左右子树交换进行reverse就得到后序遍历。

    二叉树的统一迭代法¶

    思路： 其实就是把要访问的结点放入栈中，同时将要处理的结点进行标记。其实这里就是对要处理的结点放入栈后，紧接着放一个空指针作为标记，叫做标记法。

    中序遍历：

    C++
    class solution{
    public:
    vector<int> traversal(TreeNode*root){
        stack<TreeNode*> st;//存放临时节点的栈
        vector<int> result;//存放结果
        if(root==null) return result;
        st.push(root);
        while(!st.empty()){
            TreeNode*node=st.top();//找一个指向当前栈顶的结点
            if(node!=nullptr){
                st.pop();
                if(node->right) st.push(node->right);//右
                st.push(node);//中
                st.push(null);
                if(node->left) st.push(node->left);//左
            }
            else{
                st.pop();//将null出栈
                node=st.top();//取该处理的结点
                st.pop();//弹出栈
                result.push_back(node->val);//放入结果
            }
        }
        return result;//返回结果
    }
};

    在这个代码的编写过程中，可能会出现几个难以理解的地方，

    1、为什么这个算法需要一个null来标记，同时为什么比如一开始标记的是root，最后依然能按照左跟右的顺序访问。

    因为，这个地方，比如说要访问左节点，根结点，右节点，无论怎么处理，都是从根结点进入的，所以根结点一定需要处理，但不会是最早的，所以你一开始将root标记之后，又进栈了左节点，最后一定会以左节点为基础，去进行循环，将左节点标记，从而左节点会一直在root之前，而右节点会一直在root之后，只不过这里的需要处理的结点，会逐层标记，直到二叉树没有左子树了，即访问到最深了。这时就会逆向进行处理节点，而且也是按照顺序的。

    2、为什么要弹出两次栈？ 因为，当你遇到null时，表示准备处理一个结点了，此时需要往result里面添加节点，同时不能添加当前的null，所以在这个地方进行了两次出栈，第一次是将null出栈，第二次是将需要处理的结点出栈，将其加入到result中。

    3、怎么保证循环一直向下，会执行正确？

    这个地方，我的理解是，因为他每次都是右中左的顺序入栈，每次循环会取栈顶，那么每次循环体的最后一次入栈就是栈顶元素，而每次都是左节点最后入栈，因此每次循环的下次循环都是对上一层的左子树进行处理，一直符合左中右，直到循环到最底，此时也能发现本来就应该循环到最低，从最左边的叶子结点开始访问，这个和循环的终止条件相一致，但是这个地方的终止条件是栈中元素都处理完了，而不是while(!node)，因为这个地方不止在最底层需要为空，而且在我们进行标记的时候也会触发终止条件，同时如果触发终止条件了，这个时候还不能终止循环，而是应该继续进行该结点的父节点的处理，因此这里采用了当栈为空时终止，因为栈会存放过所有结点，所有节点在未被处理之前都是不会进行出栈的，因此能够保证所有结点都处理到。同时也满足访问顺序，能够得到最终答案。

    前序遍历：

    C++
    class solution{
    public:
    vector<int> traversal(TreeNode* root){
        vector<int> result;
        stack<TreeNode*> st;
        if(root==null) return result;
        st.push(root);
        while(!st.empty()){
            TreeNode*node=st.top();
            if(!node==nullptr){
                st.pop();
                if(node->right) st.push(node->right);
                if(node->left) st.push(node->left);
                st.push(node);
                st.push(null);
            }
            else{
                st.pop();
                node=st.top();
                st.pop();
                result.push_back(node->val);
            }
        }
        return result;
    }
};

    二叉树的层序遍历¶

    这个就是层序遍历，其实层序遍历就是按照层一层一层将二叉树进行访问。

    这个地方用数据结构中的队列比较好，而用栈先进后出适合模拟深度优先遍历，也就是递归的逻辑。

    递归：其实就是在内存底部使用栈进行存储每一层函数的返回值，最后一层一层进行返回。在计算机底层递归就是利用了栈来实现的。

    层序遍历，就是图论中的广度优先遍历，只不过我们应用在二叉树上。

    代码实现：

    C++
    class solution{
    public:
    vector<vector<int>> traversal_level(TreeNode*root){
        queue<TreeNode*> que;
        if(root!=null) que.push(root);
        vector<vector<int>> result;
        while(!que.empty()){
            int size=que.size();
            vector<int> vec;
            //这个地方的size必须是当前层的结点数量；
            for(int i=0;i<size;i++){
                TreeNode* node=que.front();
                que.pop();
                vec.push_back(node->val);
                if(node->left) que.push(node->left);
                if(node->right) que.push(node->right);
            }
            result.push_back(vec);
        }
        return result;
    }
};

    递归实现：

    C++
    class solution{
    public:
    void order(TreeNode*cur,vector<vector<int>>&result,int depth){
        if(cur==nullptr) return;
        if(result.size()==depth) result.push_back(vector<int>());
        result[depth].push_bcak(cur->val);
        order(cur->left,result,depth+1);
        order(cur->right,result,depth+1);
    }
    vector<vector<int>> levelorder(TreeNode* root){
        vector<vector<int>> result;
        int depth=0;
        order(root,result,depth);
        return result;
    }
};

    这个递归实现比较绕，因此尽量掌握就行，主要就是一个递归三要素，终止条件，参数，递归体。

    二叉树的层次遍历2¶

    将二叉树的结点值自底向上进行层次遍历：

    思路：就是将正常的层序访问顺序，进行访问，然后将结果进行反转。

    C++
    class solution{
    public:
    vector<vector<int>> traversal_level(TreeNode*root){
        queue<TreeNode*> que;
        if(root!=null) que.push(root);
        vector<vector<int>> result;

            int size=que.size();

            for(int i=0;i<size;i++){
                TreeNode* node=que.front();

                if(node->left) que.push(node->left);
                if(node->right) que.push(node->right);
            }
            result.push_back(vec);
        }
        reverse(result.begin(),result.end());
        return result;
    }
};

    迭代法：

    C++
    class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        if (root == NULL) return root;
        stack<TreeNode*> st;
        st.push(root);
        while(!st.empty()) {
            TreeNode* node = st.top();              // 中
            st.pop();

    完全二叉树只有两种情况：一是满二叉树，二是最后一层叶子节点没有满。

    对于情况一，可以直接用 2^树深度 - 1 来计算，注意这里根节点深度为1。

    对于情况二，分别递归左孩子，和右孩子，递归到某一深度一定会有左孩子或者右孩子为满二叉树，然后依然可以按照情况1来计算。

    C++
    class Solution {
public:
    int countNodes(TreeNode* root) {
        if (root == nullptr) return 0;
        TreeNode* left = root->left;
        TreeNode* right = root->right;
        int leftDepth = 0, rightDepth = 0; // 这里初始为0是有目的的，为了下面求指数方便
        while (left) {  // 求左子树深度
            left = left->left;
            leftDepth++;
        }
        while (right) { // 求右子树深度
            right = right->right;
            rightDepth++;
        }
        if (leftDepth == rightDepth) {
            return (2 << leftDepth) - 1; // 注意(2<<1) 相当于2^2，所以leftDepth初始为0
        }
        return countNodes(root->left) + countNodes(root->right) + 1;
    }
};

    平衡二叉树¶

    这个题目就是给定一个二叉树，判断它是否是高度平衡的二叉树。

        if (preorderBegin == preorderEnd) return NULL;

        int rootValue = preorder[preorderBegin]; // 注意用preorderBegin 不要用0
        TreeNode* root = new TreeNode(rootValue);

        if (preorderEnd - preorderBegin == 1) return root;

        int delimiterIndex;
        for (delimiterIndex = inorderBegin; delimiterIndex < inorderEnd; delimiterIndex++) {
            if (inorder[delimiterIndex] == rootValue) break;

        // 切割中序数组
        // 中序左区间，左闭右开[leftInorderBegin, leftInorderEnd)
        int leftInorderBegin = inorderBegin;
        int leftInorderEnd = delimiterIndex;
        // 中序右区间，左闭右开[rightInorderBegin, rightInorderEnd)
        int rightInorderBegin = delimiterIndex + 1;
        int rightInorderEnd = inorderEnd;

        // 切割前序数组
        // 前序左区间，左闭右开[leftPreorderBegin, leftPreorderEnd)
        int leftPreorderBegin =  preorderBegin + 1;
        int leftPreorderEnd = preorderBegin + 1 + delimiterIndex - inorderBegin; // 终止位置是起始位置加上中序左区间的大小size
        // 前序右区间, 左闭右开[rightPreorderBegin, rightPreorderEnd)
        int rightPreorderBegin = preorderBegin + 1 + (delimiterIndex - inorderBegin);
        int rightPreorderEnd = preorderEnd;

        root->right = traversal(inorder, rightInorderBegin, rightInorderEnd, preorder, rightPreorderBegin, rightPreorderEnd);

        return root;

    654.最大二叉树¶

    C++

 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}

        int maxvalueindex=left;

                maxvalue=nums[i];
                maxvalueindex=i;
            }

        TreeNode* root=new TreeNode(0);
        root->val=maxvalue;
        root->left=digui(nums,left,maxvalueindex);
        root->right=digui(nums,maxvalueindex+1,right);
        return root;
    }
public:

        if(nums.size()==0) return nullptr;

            node->left = constructMaximumBinaryTree(newVec);

        // 最大值所在的下标右区间 构造右子树
        if (maxValueIndex < (nums.size() - 1)) {
            vector<int> newVec(nums.begin() + maxValueIndex + 1, nums.end());

        if (!root) return nullptr;

        // 处理头结点，让root移动到[L, R] 范围内，注意是左闭右闭
        while (root != nullptr && (root->val < L || root->val > R)) {

            else root = root->left; // 大于R往左走
        }
        TreeNode *cur = root;

        while (cur != nullptr) {
            while (cur->left && cur->left->val < L) {
                cur->left = cur->left->right;
            }
            cur = cur->left;
        }
        cur = root;

        // 此时root已经在[L, R] 范围内，处理右孩子大于R的情况
        while (cur != nullptr) {
            while (cur->right && cur->right->val > R) {
                cur->right = cur->right->left;
            }
            cur = cur->right;
        }
        return root;
    }

    回溯法的理论：

    其实回溯法就是递归加树，N叉树，主要把需要求解的问题集合看做一个树，每一次在递归中，for循环处理的是横向遍历，递归进行的是纵向遍历，当把所有问题集合都遍历完，这个时候递归到叶子结点，满足条件的所有结果就是答案咯。

    递归三部曲：递归函数返回值和参数，终止条件，递归循环体。

    class Solution {
private:
    vector<vector<int>> result;
    vector<int> path;
    void backtracking(vector<int>& candidates, int target, int sum, int startIndex) {
        if (sum > target) {
            return;
        }
        if (sum == target) {
            result.push_back(path);
            return;
        }

        for (int i = startIndex; i < candidates.size(); i++) {
            sum += candidates[i];
            path.push_back(candidates[i]);
            backtracking(candidates, target, sum, i); // 不用i+1了，表示可以重复读取当前的数
            sum -= candidates[i];
            path.pop_back();
        }
    }

        result.clear();
        path.clear();
        backtracking(candidates, target, 0, 0);
        return result;

    C++
    class Solution {
private:
    vector<vector<int>> result;
    vector<int> path;
    void backtracking(vector<int>& candidates, int target, int sum, int startIndex) {
        if (sum > target) {
            return;
        }


        for (int i = startIndex; i < candidates.size()&&sum+candidates[i]<=target; i++) {
            sum += candidates[i];
            path.push_back(candidates[i]);
            backtracking(candidates, target, sum, i); // 不用i+1了，表示可以重复读取当前的数
            sum -= candidates[i];
            path.pop_back();
        }
    }
public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        result.clear();
        path.clear();
        sort(candidates.begin(),candidates.end());
        backtracking(candidates, target, 0, 0);
        return result;
    }
};

    直接用回溯即可，要注意判断哪些元素已经使用过，所以需要used数组来辅助。

    C++
    class Solution {
public:
    vector<vector<int>> result;
    vector<int> path;
    void backtracking (vector<int>& nums, vector<bool>& used) {
        // 此时说明找到了一组
        if (path.size() == nums.size()) {
            result.push_back(path);
            return;
        }
        for (int i = 0; i < nums.size(); i++) {
            if (used[i] == true) continue; // path里已经收录的元素，直接跳过
            used[i] = true;
            path.push_back(nums[i]);
            backtracking(nums, used);
            path.pop_back();
            used[i] = false;
        }
    }
    vector<vector<int>> permute(vector<int>& nums) {
        result.clear();
        path.clear();
        vector<bool> used(nums.size(), false);
        backtracking(nums, used);
        return result;
    }
};
